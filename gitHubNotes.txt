cd Desktop/  		//Changes directory to Desktop
mkdir nameofFile  	//creates a new file in the directory specified
ls			// lists items in current directory
touch	fileName.txt	//creates a new file inside the current directory
open 	fileName.txt 	//opens the txt file that was just created
git init		//initializes Git repository working directory
ls -a			//shows list of all the files in directory
git status		//shows the status of git repository
git add			//pushes file to staging area
git commit -m " "	//commits file and adds message for version control
git log			//shows what changes have been made to the file
git checkout 		//rolls back to a previous version of the file
git diff <fileName> 	//shows the changes that were made to the file
